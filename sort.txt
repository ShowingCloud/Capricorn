#include<stdio.h>

void main()
{
	int i = 0, j = 0, temp;
	int num[8] = {6,8,2,7,3,5,4,1};
	for(i = 0; i < 7; i++)
	{
		for(j = 0; j < 7 - i; j++)
		{
			if(num[j] < num[j+1])
			{
				temp = num[j];
				num[j] = num[j+1];
				num[j+1] = temp;
			}
		}
	}
	for(i = 0; i < 8; i++)
	{
		printf("num[%d] = %d\t", i, num[i]);
	}


}
